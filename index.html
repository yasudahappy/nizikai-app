    <!-- Firebase連携ロジック -->

    <script type="module">

        import { initializeApp } from "https://www.gstatic.com/firebasejs/10.11.1/firebase-app.js";

        import { getFirestore, collection, addDoc, onSnapshot, query, orderBy } from "https://www.gstatic.com/firebasejs/10.11.1/firebase-firestore.js";


        // お預かりした設定情報

        const firebaseConfig = {

            apiKey: "AIzaSyBe8BmE_fzosz9jpk89FCGFQ8PHOI0MzFw",

            authDomain: "nizikai-38325.firebaseapp.com",

            projectId: "nizikai-38325",

            storageBucket: "nizikai-38325.firebasestorage.app",

            messagingSenderId: "474200648412",

            appId: "1:474200648412:web:493524949887efc0cfe219",

            measurementId: "G-MHQX4H0DLR"

        };


        const app = initializeApp(firebaseConfig);

        const db = getFirestore(app);


        // 要素の取得

        const form = document.getElementById('rsvp-form');

        const inputName = document.getElementById('input-name');

        const statusAttend = document.getElementById('status-attend');

        const statusAbsent = document.getElementById('status-absent');

        const labelAttend = document.getElementById('label-attend');

        const labelAbsent = document.getElementById('label-absent');

        const guestCountWrapper = document.getElementById('guest-count-wrapper');

        const inputGuestCount = document.getElementById('input-guest-count');

        const submitBtn = document.getElementById('submit-btn');

        const attendeesContainer = document.getElementById('attendees-container');

        const attendeesList = document.getElementById('attendees-list');

        const attendCountEl = document.getElementById('attend-count');

        const toast = document.getElementById('toast');


        // ラジオボタンの切り替えアニメーション

        function updateStyles() {

            if (statusAttend.checked) {

                labelAttend.className = "flex-1 cursor-pointer rounded-xl border-2 p-3 text-center transition-all radio-selected";

                labelAbsent.className = "flex-1 cursor-pointer rounded-xl border-2 p-3 text-center transition-all radio-unselected hover:bg-white/10";

                guestCountWrapper.style.display = 'block';

            } else {

                labelAbsent.className = "flex-1 cursor-pointer rounded-xl border-2 p-3 text-center transition-all radio-selected";

                labelAttend.className = "flex-1 cursor-pointer rounded-xl border-2 p-3 text-center transition-all radio-unselected hover:bg-white/10";

                guestCountWrapper.style.display = 'none';

            }

        }

        statusAttend.addEventListener('change', updateStyles);

        statusAbsent.addEventListener('change', updateStyles);


        function showToast(msg) {

            toast.textContent = msg;

            toast.classList.remove('hidden');

            setTimeout(() => toast.classList.add('hidden'), 3000);

        }


        // データの初期読み込み

        fetchData();


        // フォーム送信

        form.addEventListener('submit', async (e) => {

            e.preventDefault();

            if (!inputName.value.trim()) return;


            submitBtn.disabled = true;

            submitBtn.textContent = '送信中...';


            const status = statusAttend.checked ? 'attend' : 'absent';

            const count = status === 'attend' ? parseInt(inputGuestCount.value) : 0;


            try {

                // データベースの「attendances」というフォルダに保存

                await addDoc(collection(db, 'attendances'), {

                    name: inputName.value.trim(),

                    status: status,

                    guestCount: count,

                    createdAt: Date.now()

                });

                

                inputName.value = '';

                inputGuestCount.value = '1';

                showToast('出欠のお返事を送信しました！✨');

            } catch (error) {

                console.error("送信エラー:", error);

                showToast('エラーが発生しました。時間を置いてお試しください。');

            }

            

            submitBtn.disabled = false;

            submitBtn.textContent = 'お返事を送信する';

        });


        // リアルタイムデータ取得

        function fetchData() {

            const q = query(collection(db, 'attendances'), orderBy('createdAt', 'desc'));

            onSnapshot(q, (snapshot) => {

                if (snapshot.empty) {

                    attendeesContainer.classList.add('hidden');

                    return;

                }

                

                attendeesContainer.classList.remove('hidden');

                let html = '';

                let totalAttendees = 0;


                snapshot.forEach(doc => {

                    const data = doc.data();

                    if (data.status === 'attend') {

                        totalAttendees += (data.guestCount || 1);

                    }

                    

                    const badge = data.status === 'attend'

                        ? '<span class="px-2 py-0.5 rounded text-xs font-bold bg-green-100 text-green-700">出席</span>'

                        : '<span class="px-2 py-0.5 rounded text-xs font-bold bg-gray-200 text-gray-600">欠席</span>';

                    

                    const guestText = (data.status === 'attend' && data.guestCount > 1)

                        ? `<span class="text-sm font-normal text-gray-500 ml-1">(${data.guestCount}名)</span>`

                        : '';


                    // 名前を安全に表示（エスケープ）

                    const div = document.createElement('div');

                    div.textContent = data.name;

                    const safeName = div.innerHTML;


                    html += `

                        <div class="bg-gray-50 rounded-xl p-4 border border-gray-100">

                            <div class="flex items-center gap-2">

                                ${badge}

                                <span class="font-bold text-[#3d405b]">${safeName}${guestText}</span>

                            </div>

                        </div>

                    `;

                });


                attendeesList.innerHTML = html;

                attendCountEl.textContent = totalAttendees;

            }, (error) => {

                console.error("取得エラー:", error);

            });

        }
